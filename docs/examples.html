REST API with Full Features

import asyncio
from fastapi import APIRouter
from linglong_web import (
    LinglongAppServer,
    LinglongConfigBase,
    init_config,
    init_resources,
    get_db,
    get_redis,
)
from linglong_web.decorators import cacher, limiter
from linglong_web.http import LinglongHttpClient

# 1. Define Configuration
class ProdConfig(LinglongConfigBase):
    DEBUG = False
    SERVICE_NAME = "my-api"
    PGSQL_HOST = "postgres.internal"
    REDIS_HOST = "redis.internal"

# 2. Initialize
async def main():
    init_config({"prod": ProdConfig}, mode_name="prod")
    await init_resources(
        db_config={"host": "postgres.internal", "database": "mydb"},
        redis_config={"host": "redis.internal"}
    )

    # 3. Create Router
    router = APIRouter()

    @router.get("/users/{user_id}")
    @cacher(ttl=60)
    @limiter(max_calls=100, period=60)
    async def get_user(user_id: int):
        db = await get_db()
        return await db.fetchone(
            "SELECT * FROM users WHERE id = $1", user_id
        )

    # 4. Start Server
    server = LinglongAppServer()
    await server.initialize(
        service_name="my-api",
        router=router,
        config_dict={"prod": ProdConfig},
    )
    await server.start(host="0.0.0.0", port=8080)

asyncio.run(main())

Microservice with HTTP Client

from linglong_web.http import LinglongHttpClient

client = LinglongHttpClient(
    base_url="https://api.example.com",
    timeout=30
)

async def fetch_user_data(user_id: int):
    # Request-Id is automatically propagated
    response = await client.get(
        f"/users/{user_id}",
        headers={"X-Custom-Header": "value"}
    )
    return response.json()

Scheduled Background Task

from aioclock import Aioclock
from linglong_web import get_redis

app = Aioclock()

@app.cron("0 2 * * *")
async def daily_cleanup():
    redis = await get_redis()
    await redis.delete("cache:expired:*")
    print("Daily cleanup completed")
